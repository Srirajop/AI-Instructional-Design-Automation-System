Tables: ['projects', 'users']

  business_unit: VARCHAR(255)
  design_doc: TEXT
  storyboard: TEXT
  created_at: DATETIME

Users columns:
  id: INTEGER
  name: VARCHAR(100)
  email: VARCHAR(100)
  hashed_password: VARCHAR(255)
  created_at: DATETIME
